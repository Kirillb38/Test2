#include <iostream>
#include <cmath>

using namespace std;

int main() {
    double x = 1.0;
    double endX = 3.0;
    double step = 0.2;

    cout << "x\tY" << endl;
    cout << "--------------------" << endl;

    do {
        double Y = cos(x) + log(x) - exp(x);
        cout << x << "\t" << Y << endl;
        x += step;
    } while (x <= endX);

    return 0;
}
